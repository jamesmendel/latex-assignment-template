% ----------------------------------------------------------------
% Create background grid of dots
% ----------------------------------------------------------------
\definecolor{dotcolor}{RGB}{150,150,150}
\backgroundsetup{
contents={%   
  \begin{tikzpicture}
    \foreach \xdot in {0,...,34}{
        \foreach \ydot in {0,...,44}{
            \fill[color=dotcolor] (\xdot,\ydot)  circle[radius=.85pt];
        } 
    }
  \end{tikzpicture}%
},
scale=.65,
angle=0,
opacity=1
}
