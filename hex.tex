% ----------------------------------------------------------------
% Create background grid of hexagons using tikzlibrary shapes.geometric
% ----------------------------------------------------------------

% Color of hexagons -- change this color to change the background color
\definecolor{bgcolor}{RGB}{210,210,210}

\backgroundsetup{
contents={%
    \begin{tikzpicture}[x=7.5mm,y=4.34mm]
      % some styles
      \tikzset{
        box/.style={
          color=bgcolor,
          regular polygon,
          regular polygon sides=6,
          minimum size=10mm,
          inner sep=0mm,
          outer sep=0mm,
          rotate=0,
        draw
        }
      }%
    \foreach \i in {0,...,22} 
        \foreach \j in {0,...,48} {
                \node[box] at (2*\i,2*\j) {};
                \node[box] at (2*\i+1,2*\j+1) {};
            }%
    \end{tikzpicture}
},
scale=.65,
angle=0,
opacity=1
}
